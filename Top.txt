module tlc_top(
  input clk,
  input reset,
  output red,
  output yellow,
  output green
);
  
  wire count_done,reset_counter;
  wire [4:0] counts;
  
  tlc_controller controller(
    .clk(clk),
    .reset(reset),
    .count_done(count_done),
    .red(red),
    .yellow(yellow),
    .green(green),
    .reset_counter(reset_counter)
    .counts(counts)
  );
  
  
  tlc_dp datapath(
    .clk(clk),
    .reset_counter(reset_counter),
    .counts(counts),
    .count_done(count_done)
  );

endmodule
